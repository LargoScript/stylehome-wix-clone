import os

def list_files(startpath):
    # –ü–∞–ø–∫–∏, —è–∫—ñ –º–∏ —ñ–≥–Ω–æ—Ä—É—î–º–æ, —â–æ–± –Ω–µ –∑–∞–±–∏–≤–∞—Ç–∏ –≤–∏–≤—ñ–¥
    exclude = {'.git', 'node_modules', 'target', '.idea', 'venv', '__pycache__'}
    
    print(f"–°—Ç—Ä—É–∫—Ç—É—Ä–∞ –ø—Ä–æ–µ–∫—Ç—É –≤: {os.path.abspath(startpath)}\n")
    
    for root, dirs, files in os.walk(startpath):
        # –§—ñ–ª—å—Ç—Ä—É—î–º–æ –ø–∞–ø–∫–∏
        dirs[:] = [d for d in dirs if d not in exclude]
        
        level = root.replace(startpath, '').count(os.sep)
        indent = ' ' * 4 * (level)
        print(f"{indent}üìÇ {os.path.basename(root)}/")
        
        subindent = ' ' * 4 * (level + 1)
        for f in files:
            # –ü–æ–∫–∞–∑—É—î–º–æ –ª–∏—à–µ —Ü—ñ–∫–∞–≤—ñ –Ω–∞–º —Ç–∏–ø–∏ —Ñ–∞–π–ª—ñ–≤
            if f.endswith(('.xml', '.sh', '.json', '.env', '.properties', '.java', '.js', '.ts', '.html')):
                print(f"{subindent}üìÑ {f}")

if __name__ == "__main__":
    list_files('.')
